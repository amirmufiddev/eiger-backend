## 3. Implementation

### Step 1: Create WalletRepository

```typescript
// src/modules/wallet/repository/wallet.repository.ts
import { Injectable, Inject } from '@nestjs/common';
import { eq } from 'drizzle-orm';
import { DATABASE_CONNECTION } from '../../../infrastructure/database/database.module';
import { wallets } from '../../../infrastructure/database/schema';
import type { PostgresJsDatabase } from 'drizzle-orm/postgres-js';
import type * as schema from '../../../infrastructure/database/schema';

export interface WalletEntity {
  id: string;
  userId: string;
  balance: number;
}

export interface TopupData {
  amount: number;
}

@Injectable()
export class WalletRepository {
  constructor(
    @Inject(DATABASE_CONNECTION)
    private db: PostgresJsDatabase<typeof schema>,
  ) {}

  private mapToEntity(w: typeof wallets.$inferSelect): WalletEntity {
    return {
      id: w.id,
      userId: w.userId,
      balance: parseFloat(w.balance),
    };
  }

  async findByUserId(userId: string): Promise<WalletEntity | null> {
    const result = await this.db
      .select()
      .from(wallets)
      .where(eq(wallets.userId, userId))
      .limit(1);
    return result.length > 0 ? this.mapToEntity(result[0]) : null;
  }

  async findAll(): Promise<WalletEntity[]> {
    const result = await this.db.select().from(wallets);
    return result.map((w) => this.mapToEntity(w));
  }

  async updateBalance(userId: string, newBalance: number): Promise<WalletEntity | null> {
    const [result] = await this.db
      .update(wallets)
      .set({ balance: newBalance.toString(), updatedAt: new Date() })
      .where(eq(wallets.userId, userId))
      .returning();
    return result ? this.mapToEntity(result) : null;
  }

  async addBalance(userId: string, amount: number): Promise<WalletEntity | null> {
    const wallet = await this.findByUserId(userId);
    if (!wallet) return null;
    const newBalance = wallet.balance + amount;
    return this.updateBalance(userId, newBalance);
  }
}
```

### Step 2: Create WalletService

```typescript
// src/modules/wallet/services/wallet.service.ts
import { Injectable, NotFoundException, BadRequestException } from '@nestjs/common';
import { WalletRepository, WalletEntity } from '../repository/wallet.repository';

@Injectable()
export class WalletService {
  constructor(private readonly repository: WalletRepository) {}

  async getBalance(userId: string): Promise<WalletEntity> {
    const wallet = await this.repository.findByUserId(userId);
    if (!wallet) throw new NotFoundException('Wallet not found');
    return wallet;
  }

  async getAllWallets(): Promise<WalletEntity[]> {
    return this.repository.findAll();
  }

  async topup(userId: string, amount: number): Promise<WalletEntity> {
    if (amount <= 0) {
      throw new BadRequestException('Topup amount must be positive');
    }
    const result = await this.repository.addBalance(userId, amount);
    if (!result) throw new NotFoundException('Wallet not found');
    return result;
  }
}
```

### Step 4: Create MemberController

```typescript
// src/modules/wallet/controllers/wallet.member.controller.ts
import {
  Controller,
  Get,
  Post,
  Body,
  UseGuards,
} from '@nestjs/common';
import {
  ApiTags,
  ApiOperation,
  ApiResponse,
  ApiBearerAuth,
} from '@nestjs/swagger';
import { WalletService } from '../services/wallet.service';
import { AuthGuard } from '../../../common/guards/auth.guard';
import { CurrentUser, CurrentUserData } from '../../../common/decorators/current-user.decorator';
import { WalletResponseDto, TopupRequestDto } from '../dto';

@ApiTags('Wallet')
@Controller('wallet')
@UseGuards(AuthGuard)
@ApiBearerAuth()
export class WalletMemberController {
  constructor(private readonly service: WalletService) {}

  @Get('balance')
  @ApiOperation({ summary: 'Get wallet balance' })
  @ApiResponse({ status: 200, type: WalletResponseDto })
  async getBalance(@CurrentUser() user: CurrentUserData): Promise<WalletResponseDto> {
    return this.service.getBalance(user.id);
  }

  @Post('topup')
  @ApiOperation({ summary: 'Topup wallet' })
  @ApiResponse({ status: 200, type: WalletResponseDto })
  async topup(
    @CurrentUser() user: CurrentUserData,
    @Body() dto: TopupRequestDto,
  ): Promise<WalletResponseDto> {
    return this.service.topup(user.id, dto.amount);
  }
}
```

### Step 5: Create AdminController

```typescript
// src/modules/wallet/controllers/wallet.admin.controller.ts
import {
  Controller,
  Get,
  Post,
  Param,
  Body,
  UseGuards,
} from '@nestjs/common';
import {
  ApiTags,
  ApiOperation,
  ApiResponse,
  ApiBearerAuth,
} from '@nestjs/swagger';
import { WalletService } from '../services/wallet.service';
import { AuthGuard } from '../../../common/guards/auth.guard';
import { RolesGuard } from '../../../common/guards/roles.guard';
import { Roles } from '../../../common/decorators/roles.decorator';
import { WalletResponseDto, TopupRequestDto } from '../dto';

@ApiTags('Admin - Wallet')
@Controller('admin/wallet')
@UseGuards(AuthGuard, RolesGuard)
@Roles('admin')
@ApiBearerAuth()
export class WalletAdminController {
  constructor(private readonly service: WalletService) {}

  @Get()
  @ApiOperation({ summary: 'Get all wallets' })
  @ApiResponse({ status: 200, type: [WalletResponseDto] })
  async getAll(): Promise<WalletResponseDto[]> {
    return this.service.getAllWallets();
  }

  @Get(':userId')
  @ApiOperation({ summary: 'Get user wallet by userId' })
  @ApiResponse({ status: 200, type: WalletResponseDto })
  async getByUserId(@Param('userId') userId: string): Promise<WalletResponseDto> {
    return this.service.getBalance(userId);
  }

  @Post(':userId/topup')
  @ApiOperation({ summary: 'Topup user wallet' })
  @ApiResponse({ status: 200, type: WalletResponseDto })
  async topup(
    @Param('userId') userId: string,
    @Body() dto: TopupRequestDto,
  ): Promise<WalletResponseDto> {
    return this.service.topup(userId, dto.amount);
  }
}
```
